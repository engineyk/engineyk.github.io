1.1 微平面（Microfacet）

  • 光线的反射

    • 镜面反射光(黄线): 直接从表面反射出去，不进入物体内部。

1.2 能量守恒

  • 入射光的总能量 < 出射光的总能量
  • 最终颜色 = 漫反射 + 镜面反射 + GI <= 1
  • 光能传播过程中的能量守衡体现在三个方面。

    • 一个对象反射出来的光照信息，不可能超过它接受到的信息。也就是说，全反射是一个物体的极限。

    • 一个物体越光亮，那么它的颜色信息应该越少。（refection和diffuse是一个插值关系）

      oneMinusReflectivity = 1 - lerp(dielectricSpec, 0, metallic)

diffColor = albedo * (oneMinusReflectivity)

    • 一个物体越平滑，那么它的高亮点会越小，越亮。

2. 基于物理的BRDF（Use a physically based BRDF)

BRDF（Bidirectional Reflectance Distribution Function，双向反射分布函数）

f(l,v)=dLo(v)dE(l)f(l, v) = \frac{dL_o(v)}{dE(l)}f(l,v)=dE(l)dLo​(v)​

其中：

  • f(l,v)f(l, v)f(l,v)：BRDF函数，表示从入射方向 lll 到出射方向 vvv 的反射属性
  • dLo(v)dL_o(v)dLo​(v)：出射方向 vvv 上的辐射亮度微分
  • dE(l)dE(l)dE(l)：入射方向 lll 上的照度微分

（1）概念理解

  • 入射光 l 照到物体表面上，反射光线为v，那么反射光的亮度（辐射率）和入射光的能量(辐照度)会成一个比例，而这个比例就是BDRF。
  • 可以理解为，在某一个特定的角度观看某个点时，各个方向的入射光对该点的最终光亮度产生的贡献比例，其实它就是一些零点几的小数。

（2）为什么是辐射率和辐射度微分的比值，而不是辐射率和辐射率，或者辐照度和辐照度的微分比值？
见参考资料的两种解释，比较好理解的是从数学角度来说，只有用Irradiance的微分，这个比值才会有意义，否则直接用出射光的dLo比上入射光的L，会趋于0，如果是dE，则是Li * cosθ * dw，也是趋于0，是有意义的。

描述物体表面如何反射光线的方程
表示了当给定一条入射光的时候，某一条特定的出射光线的性质是怎么样的。
它的精确定义是出射光辐射率(Radiance)的微分和入射光辐照度(Irradiance)的微分之比。

float3 BRDF(float3 v, float3 l, float3 n, float3 f0, float roughness) {
    float h  = normalize(l + v); // 光源和视角的中间
    float nh  = max(dot(n, h), 0.0);
    float nl  = max(dot(n, v), 0.0);
    // 计算法线分布函数
    float D = D_GGX(n, h, roughness);
    // 计算菲涅尔函数
    float3 F = F_Schlick(v, h, f0);
    // 计算几何遮挡函数
    float G = G_SchlickGGX(n, v, l, roughness);
    // 计算BRDF
    float3 brdf = (D * G * F) / (4.0 * nv * nl);
    return brdf;
}

// Trowbridge-Reitz GGX
// 粗糙的物体a=1：1
// 光滑的物体a=0：无穷小
// Epic GLSL
float DistributionGGX(vec3 N, vec3 H, float roughness)
{
    float a   = roughness * roughness; // a:粗糙度
    float a2  = a * a;
    float nh  = max(dot(N, H), 0.0);
    float nh2 = nh*nh;
    float d = (nh2 * (a2 - 1.0) + 1.0);
    return a2 / (PI * d * d);
}

// GGX（Trowbridge-Reitz）
float D_GGX(float a2, float nh) // a:粗糙度
{
  float d = (nh * a2 - nh) * nh + 1;  // 2 mad
  return a2 / (PI * d * d);           // 4 mul, 1 rcp
}

2.2.2 菲涅尔（F）Fresnel-Schlick近似法(Approximation)

  • 不同角度有不同强度的反射光
  • 水的材质现象,材质表面对于光线的反射
    • N和V的角度
    • 正交时当光线垂直入射的时候（沿着法线入射的时候）是最弱的，
    • 非垂直而当光线与法线的夹角逐渐增大，则反射也会增强。
  • 类似物理碰撞，如光线是子弹，如水是钢盔
  • 如果是正面，冲撞最大，直接穿透。
  • 如果是侧面，冲撞最小，会反弹。
  • 纯金属的反射率F0为其颜色，纯金属的颜色因为反射率不同造成
  • 非金属的反射率F0为其明暗，非金属的颜色因为不同波长的光的吸收率不同造成

正交时,夹角为90°, 视线V和法线夹角为0时，hv=1，代入公式，二项式为0，值为F0
非垂直,夹角<90°, 视线V和法线夹角为1时，hv=0，代入公式，二项式为0，值为0

float GeometrySmith(vec3 N, vec3 V, vec3 L, float roughness) {
  float nv = max(dot(N, V), 0.0);
  float nl = max(dot(N, L), 0.0);
  float ggx1 = SchlickGGX(nl, roughness); // 几何阴影
  float ggx2 = SchlickGGX(nv, roughness); // 几何遮挡
  return ggx1 * ggx2;
}

float SchlickGGX(float nv, float roughness) {
  float r = roughness + 1.0;
  float k = (r * r) / 8.0;
  float denom = nv * (1.0 - k) + k;
  return nv / denom;
}

3. DualSpecularGGX 

vec3 DualSpecularGGX(float NoL, float NoV, float NoH, float LoH, float AverageRoughness, vec3 SpecularColor)
{
  //AverageRoughness = GBuffer.Roughness
  float AverageToRoughness0 = DUAL_SPECULAR.x * 2.0;
  float AverageToRoughness1 = DUAL_SPECULAR.y * 2.0;

  float Lobe0Roughness = max(clamp((AverageRoughness * AverageToRoughness0), 0.0, 1.0), 0.02);
  float Lobe1Roughness = clamp((AverageRoughness * AverageToRoughness1), 0.0, 1.0);

  float AverageAlpha2 = pow((AverageRoughness), 2.0);
  float Lobe0Alpha2 = pow((Lobe0Roughness), 2.0);
  float Lobe1Alpha2 = pow((Lobe1Roughness), 2.0);

  // Generalized microfacet specular
  float D = mix(GGXTermNotNormalized(NoH, Lobe0Alpha2), GGXTermNotNormalized(NoH, Lobe1Alpha2), DUAL_SPECULAR.z);
  float Vis = SmithJointGGXVisibilityTerm(NoL, NoV, AverageAlpha2); // Average visibility well approximates using two separate ones (one per lobe).
  vec3 F = FresnelTerm(SpecularColor, LoH);
  return (D * Vis) * F;
}

float GGXTermNotNormalized (float NdotH, float roughness)
{
    float a2 = roughness * roughness;
    float d = (NdotH * a2 - NdotH) * NdotH + 1.0; // 2 mad
    return min(a2 / (d * d + 1e-5), 128.0);
    // This function is not intended to be running on Mobile,
    // therefore epsilon is smaller than what can be represented by half
}
float SmithJointGGXVisibilityTerm(float nl, float nv, float roughness)
{
    float a = roughness;
    float lambdaV = nl * (nv * (1.0 - a) + a);
    float lambdaL = nv * (nl * (1.0 - a) + a);
    return min(0.5 / (lambdaV + lambdaL + 1e-5), 128.0);
}

vec3 FresnelTerm (vec3 F0, float cosA)
{
    float t = Pow5 (1.0 - cosA);   // ala Schlick interpoliation
    return F0 + (1-F0) * t;
}

6. 光照实现 

void main() 
{
  vec3 Lo = vec3(0.0);
  for(int i = 0; i < 4; ++i) 
  {
    // wi
    vec3 L = normalize(lightPositions[i] - WorldPos);
    vec3 H = normalize(V + L);

    float distance    = length(lightPositions[i] - WorldPos);
    // 计算光源在点fragPos的衰减系数
    float attenuation = 1.0 / (distance * distance);    // 衰减
    vec3 radiance     = lightColors[i] * attenuation;   // 辐射

    vec3 F0 = vec3(0.04); 
    F0 = mix(F0, albedo, metallic);
    vec3 F = fresnelSchlick(max(dot(H, V), 0.0), F0);

    float NDF = DistributionGGX(N, H, roughness);       
    float G   = GeometrySmith(N, V, L, roughness);

    vec3 numerator    = NDF * G * F;
    float denominator = 4.0 * max(dot(N, V), 0.0) * max(dot(N, L), 0.0); 
    vec3 specular     = numerator / max(denominator, 0.001);  // 为了防止除0情况而特意加上的。

    vec3 kS = F;
    vec3 kD = vec3(1.0) - kS;
      
    kD *= 1.0 - metallic; // 由于金属表面不折射光，没有漫反射颜色，通过归零kD来实现这个规则


    const float PI = 3.14159265359;

    float NdotL = max(dot(N, L), 0.0);    
    // float cosTheta    = max(dot(N, Wi), 0.0);
    // Lo出射辐射度（Radiosity）  反射方程在半球Ω的积分∫的结果  
    Lo += (kD * albedo / PI + specular) * radiance * NdotL;
  }
  vec3 ambient = vec3(0.03) * albedo * ao;
  vec3 color = ambient + Lo;  
}

vec3 fresnelSchlick(float cosTheta, vec3 F0)
{
  return F0 + (1.0 - F0) * pow(1.0 - cosTheta, 5.0);
} 

float DistributionGGX(vec3 N, vec3 H, float roughness)
{
  float a      = roughness*roughness;
  float a2     = a*a;
  float NdotH  = max(dot(N, H), 0.0);
  float NdotH2 = NdotH*NdotH;

  float num   = a2;
  float denom = (NdotH2 * (a2 - 1.0) + 1.0);
  denom = PI * denom * denom;

  return num / denom;
}

float GeometrySchlickGGX(float NdotV, float roughness)
{
  float r = (roughness + 1.0);
  float k = (r*r) / 8.0;

  float num   = NdotV;
  float denom = NdotV * (1.0 - k) + k;

  return num / denom;
}

float GeometrySmith(vec3 N, vec3 V, vec3 L, float roughness)
{
  float NdotV = max(dot(N, V), 0.0);
  float NdotL = max(dot(N, L), 0.0);
  float ggx2  = GeometrySchlickGGX(NdotV, roughness);
  float ggx1  = GeometrySchlickGGX(NdotL, roughness);

  return ggx1 * ggx2;
}
